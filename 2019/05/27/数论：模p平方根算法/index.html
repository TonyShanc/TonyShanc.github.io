数论 模p平方根算法（C）


模p平方根算法用来快速计算二次同余式

计算模p平方根例子:
$x^2$ $\equiv$ 4 (mod 281)
该二次同余式解为 x $\equiv$ 132 (mod 281)

#include <stdio.h>
#include <stdlib.h>
#define MAX 10
int cal_s(int p)
{
    int temp=p-1;
    while(temp%2==0)
    {
        temp = temp/2;
    }
    return temp;

}
int inverse(int a,int m)
{
    for(int i=1;i<m;i++)
        if((a*i)%m==1)return i;
}
int cal_t(int p)
{
    int temp=p-1;
    int cnt=0;
    while(temp%2==0)
    {
        temp = temp/2;
        cnt++;
    }
    return cnt;
}
int getn(int p)//get模的平方非剩余n
{
    for(int i=2;i<p;i++)
    {
        if(power(i,(p-1)/2,p)==p-1)
            return i;
    }
    return -1;
}

int  power(int x,int y,int m)//计算 x^y%m
{
    if(y==0)return 1%m;
    int  ret=x%m;
    for(int i=0;i<y-1;i++)
        ret=(ret*x)%m;
    return ret;
}

int normal_power(int a,int b)//计算a^b
{
    int ret=1;
    for(int i=0;i<b;i++)
        ret*=a;
    return ret;
}

int is_prime(int n)
{
    if(n==2)return 0;
    double x=sqrt(n);
    for(int i=2;i<=x;i++)
        if(n%i==0)
            {
                return 0;
            }
    return 1;
}
int main()
{
    int a,p,n,b,s,t,_a;
    int x[MAX];
    printf("请输入a,p, 用空格区分：");
    scanf("%d %d",&a, &p);
    if(!is_prime(p)){
            printf("p不是素数，请重新尝试\n");
            return 0;
    }
    printf("p是素数\n");
    if(power(a,(p-1)/p,p)==p-1){
        printf("该二次同余式无解\n");
        return 0;
    }
    printf("该二次同余式有解\n");
    n = getn(p);
    printf("模p平方二次非剩余n = %d\n",n);
    while(a<0)a+=p;
    while(a>p)a-=p;
    s = cal_s(p);
    t = cal_t(p);
    if(t==1)
    {
        x[0] = power(a,(s+1)/2,p);
        printf("由于t = 1，同余式的解 = %d\n",x[0]);
        return 0;
    }
    b = power(n,s,p);
    _a = inverse(a,p);
    printf("a = %d, p = %d, n = %d, t = %d, s = %d, b = %d\n",a,p,n,t,s,b);
    for(int i=0;i<MAX;i++)x[i]=0;
    x[t-1]=power(a,(s+1)/2,p);
    printf("x[%d] = %d\n",t-1,x[t-1]);
    for(int k=1;k<=t-1;k++)
    {
        int judge=power((_a*x[t-k]*x[t-k])%p,normal_power(2,t-k-1),p);
        if(judge==1)
        {
            x[t-k-1] = x[t-k];
            printf("j[%d] = 0\n",k-1);
            printf("x[%d] = %d\n", t-k-1,x[t-k-1]);
        }
        else if(judge==p-1)
        {
            x[t-k-1]=(power(b,normal_power(2,k-1),p)*x[t-k])%p;
            printf("j[%d] = 1\n",k-1);
            printf("x[%d] = %d\n", t-k-1, x[t-k-1]);
        }
    }
    printf("二次同余式解x[0] = %d\n",x[0]);
    return 0;
}
