go context.ValueCtxзҡ„key

WithValue returns a copy of parent in which the value associated with key isval. Use context Values only for request-scoped data that transits processes and APIs, not for passing optional parameters to functions. The provided key must be comparable and should not be of type string or any other built-in type to avoid collisions between packages using context. Users of WithValue should define their own types for keys. To avoid allocating when assigning to an interface{}, context keys often have concrete type struct{}. Alternatively, exported context key variablesвҖҷ static type should be a pointer or interface.

//util/ctxutil.go

package mycontext
// unexported, comparable
var key = &struct{}

func WithTags(ctx context.Context, tags map[string]string) context.Context{
    if ctx == nil {
        ctx = context.Background()
    }
    return context.WithValue(ctx, ctxMarkerKey, tags)
}
func Extract(ctx context.Context)map[string]string{
  // ctx is ptr
  if ctx == nil {
    return make(map[string]string)
  }
  if value, ok := ctx.Value(key).(map[string]string); ok {
    return value
  }else {
    return make(map[string]string)
  }
}
