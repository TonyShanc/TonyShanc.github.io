go 类型的可比较性

111

go内置类型都是可比较的，类型会直接比较内容，指针类型先判断指针指向地址再判断内容
常用复合类型的可比较性比较复杂，demo如下

func isComparable(a interface&#123;&#125;) &#123;
    fmt.Println(reflect.ValueOf(a).Type().Comparable())
&#125;
type Dog struct &#123;
    name string
&#125;
func (d Dog) eat()&#123;
    fmt.Println("dog " + d.name + " eat")
&#125;
type Cat struct &#123;
    name string 
    fur
&#125;
type fur struct&#123;
    color string
    num   int64
&#125;
func (c Cat) eat() &#123;
    fmt.Println("cat " + c.name + " eat")
&#125;
type Animal interface &#123;
    eat()
&#125;
type AnimalB interface &#123;
  eat()
&#125;
func main()&#123;
  a := make(map[string]int)
  isComparable(a)           // false

  b := []int64&#123;1,2&#125;
  isComparable(a)           // false

  c := struct &#123;
      id int
      name string
  &#125;&#123;
      1,
      "sc",
  &#125;
  isComparable(c)           // true

  d := struct &#123;
      id int
      name string
      clildrenId []int
  &#125;&#123;
      1,
      "shanchao",
      []int &#123;1,2,3,4&#125;,
  &#125;
  isComparable(d)            // false struct内部存在不可比较类型slice

  var(
    d1 = Dog&#123;name: "liuyang"&#125;
    d2 = Dog&#123;name: "liuyang"&#125;
    dog Animal = Dog&#123;name: "liuyang"&#125;
    dog2 Animal = Dog&#123;name: "liuyang"&#125;
    dogPtr Animal = &Dog&#123;name: "liuyang"&#125;
    dogB AnimalB = Dog&#123;name: "liuyang"&#125;
    dogBPtr AnimalB = &Dog&#123;name: "liuyang"&#125;
    cat Animal = Cat&#123;name: "liuyang"&#125;
    cat2  = Cat&#123;name: "sc", fur: fur&#123;color: "yellow", num: int64(1<<56),&#125;,&#125;
    cat3  = Cat&#123;name: "sc", fur: fur&#123;color: "blue", num: int64(1<<56),&#125;,&#125;
  )
  isComparable(d)            // true
  isComparable(dog)          // true
  isComparable(dogB)         // true
  isComparable(cat)          // true
  fmt.Println(d1 == d2)      // true 
  fmt.Println(dog == dogPtr) // false
  fmt.Println(dog == dog2)   // true 
  fmt.Println(dog == dogB)   // true 比较与接口无关
  fmt.Println(dogPtr == dogBPtr) // false 先指针指向地址是否相同
  fmt.Println(dog == cat)    // false 考虑了自定义类型
  fmt.Println(dog == cat)    // false 考虑了自定义类型
  fmt.Println(cat2 == cat3)  // false 递归判断
&#125;

可见map, slice是不可比较的；比较interface无关；对自定义类型struct的比较先判断是否是指针，再判断属性类型、属性值，具有递归判断的性质; 嵌入不可比较类型如map slice的struct不可比较，编译会报错；

另一个有趣的点涉及到interface{}:

type myString string
func compare(a interface&#123;&#125;) bool &#123;
    if interface&#123;&#125;("haha") == a &#123;
        return true
    &#125;
    return false
&#125;

func main()&#123;
    var (
        s1 = myString("haha")
        s2 = "haha"
    )
    fmt.Println(s1 == "haha")    // true
    fmt.Println(compare(s1))     // false
    fmt.Println(compare(s2))     // true
&#125;

自定义类型与某个基础内嵌类型相同时用==不会考虑自定义的类型，而interface{}类型相互比较时，会先比较转为interface{}前的自定义类型，之后才会进行值比较。 interface{}这个特性被contex.ValueCtx用于key判断。
